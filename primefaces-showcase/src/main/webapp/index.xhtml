Quick Start

So simple to use

Add the Maven dependency, declare the namespace, drop a component tag — you have a production-ready UI. No complex setup, no boilerplate ceremony.

Read the Docs
pom.xml page.xhtml 
<dependency>
    <groupId>org.primefaces</groupId>
    <artifactId>primefaces</artifactId>
    <version>#{app.primeFacesVersion}</version>
    <classifier>jakarta</classifier>
</dependency>

<!-- Optional: PrimeFlex CSS utility -->
<dependency>
    <groupId>org.webjars.npm</groupId>
    <artifactId>primeflex</artifactId>
    <version>4.0.0</version>
</dependency>
<html xmlns="http://www.w3.org/1999/xhtml"
      xmlns:h="jakarta.faces.html"
      xmlns:p="primefaces">
<h:head>
    <title>My App</title>
</h:head>
<h:body>
    <p:dataTable var="item" value="\#{bean.items}">
        <p:column headerText="Name">\#{item.name}</p:column>
        <p:column headerText="Status">
            <p:tag value="\#{item.status}" />
        </p:column>
    </p:dataTable>
</h:body>
</html>
